CREATE OR REPLACE PROCEDURE FIND_MIN_MAX_SALARY (
    P_DEPARTMENT_NAME IN VARCHAR2,
    P_MIN_SALARY OUT NUMBER,
    P_MAX_SALARY OUT NUMBER
)
IS
BEGIN
    -- Fetch min and max salary for the given department
    SELECT MIN(SALARY), MAX(SALARY)
    INTO P_MIN_SALARY, P_MAX_SALARY
    FROM EMPLOYEES
    WHERE DEPARTMENT_NAME = P_DEPARTMENT_NAME;

EXCEPTION
    WHEN NO_DATA_FOUND THEN
        P_MIN_SALARY := NULL;
        P_MAX_SALARY := NULL;
        DBMS_OUTPUT.PUT_LINE('No employees found in the given department.');
    WHEN OTHERS THEN
        DBMS_OUTPUT.PUT_LINE('An error occurred: ' || SQLERRM);
END FIND_MIN_MAX_SALARY;
/

execute
DECLARE
    V_MIN_SALARY NUMBER;
    V_MAX_SALARY NUMBER;
BEGIN
    -- Calling the procedure for the 'IT' department
    FIND_MIN_MAX_SALARY('IT', V_MIN_SALARY, V_MAX_SALARY);
    
    -- Displaying output values
    DBMS_OUTPUT.PUT_LINE('Min Salary: ' || V_MIN_SALARY);
    DBMS_OUTPUT.PUT_LINE('Max Salary: ' || V_MAX_SALARY);
END;
